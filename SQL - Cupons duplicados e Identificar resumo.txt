--IDENTIFICA CUPONS DUPLICADOS

SELECT
  DOCNUMDOCFMT AS SEQUENCIAL,
  CXANUM AS CAIXA,
  DOCVLRTOT AS VALOR
FROM V_SAIDA
WHERE LOJCOD = '000001'
AND CXANUM = '002'
AND DOCDATEMI = '09/01/2019'
AND DOCTIP = 'NFCE'
AND DOCSTA = 'E'
GROUP BY DOCNUMDOCFMT,
		 CXANUM,
         DOCVLRTOT
HAVING COUNT(*) > 1
ORDER BY 3


--IDENTIFICA CCL DO LANÇAMENTO PARA ENCONTRAR O RESUMO
SELECT
  AGECOD,
  CCLCOD,
  MEIAUTCOD,
  *
FROM CARTAO_CREDITO_LANCAMENTO
WHERE LOJCOD = '000001'
AND CCLDATEMI = '10/05/2018'
AND CCLVLRNOM = '6.0000'

  AGECOD,
  *
FROM ITEM_RESUMO_CARTAO_CREDITO
