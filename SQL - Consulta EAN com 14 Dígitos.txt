select
 r.REFCOD,r.REFDES,e.EANCOD 
from
 EAN E 
 inner join REFERENCIA r on e.REFPLU = r.REFPLU
where
 EXISTS
   (
      select 
        REFPLU, 
        COUNT(*) 
      from 
        EAN 
      WHERE
        E.REFPLU = EAN.REFPLU
      group by 
       REFPLU 
      having count(*) = 1
 ) AND
 len(e.EANCOD) = 14
